cmake_minimum_required(VERSION 2.6)

PROJECT(VTKPCLConversions)

FIND_PACKAGE(VTK REQUIRED)
INCLUDE(${VTK_USE_FILE})

ADD_EXECUTABLE(VTKtoPCL VTKtoPCL.cxx)
TARGET_LINK_LIBRARIES(VTKtoPCL ${VTK_LIBRARIES} ${PCL_COMMON_LIBRARIES} ${PCL_IO_LIBRARIES})

ADD_EXECUTABLE(PCLtoVTK PCLtoVTK.cxx)
TARGET_LINK_LIBRARIES(PCLtoVTK ${VTK_LIBRARIES} ${PCL_COMMON_LIBRARIES} ${PCL_IO_LIBRARIES})
